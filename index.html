Demos

  • val fileName = "file.scala"
val lineNumber = 37

scalaVirtualMachine.getOrCreateBreakpointRequest(
  fileName,
  lineNumber
).foreach(breakpointEvent => {
  val f = breakpointEvent.fileName
  val l = breakpointEvent.lineNumber

  println(s"Reached breakpoint $f:$l")
})
